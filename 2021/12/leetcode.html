func lengthOfLongestSubstring(s string) int {
    if len(s) == 0 {
        return 0
    }

    hash := make([]int, 127)
    for i, _ := range hash {
        hash[i] = -1
    }

    start, sz := -1, 0
    maxLen := 1
    for i := 0; i < len(s); i++ {
        j := hash[int(s[i])]
        if j > start {
            start = j
        }

        sz = i - start

        if sz > maxLen {
            maxLen = sz
        }

        hash[int(s[i])] = i
    }

    return maxLen
}

func reverse(x int) int {
	var max, ret, r int64
	sign, xx := int64(1), int64(xx)

	if xx > 0 {
		max = int64(math.MaxInt32)
	} else {
		max = int64(-math.MinInt32)
		sign = -1
		xx = int64(-x)
	}

	for xx > 0 {
		r = xx % 10
		xx = xx/10
		ret = ret * 10 + r
		if ret > max {
			return 0
		}
	}

	return int(sign*ret)
}

// 用行数组模拟 Z 形
func convert(s string, numRows int) string {
    if numRows <= 1 {
        return s
    }

    var tmp [][]byte
    tmp = make([][]byte, numRows)
    for i := 0; i < numRows; i++ {
        tmp[i] = []byte{}
    }

    row := 0
    down := 1
    down2up := numRows - 1
    for _, c := range s {
        tmp[row] = append(tmp[row], byte(c))
        switch {
            case row == 0:
            down = 1
            case row == down2up:
            down = -1
            default:
        }

        row += down
    }

    var result []byte
    for _, rows := range tmp {
        result = append(result, rows...)
    }

    return string(result)
}

// 另外，最通用的方法应该有限状态机（自动机）
func myAtoi(s string) int {
  abs, sign, i, n := int64(0), int64(1), 0, len(s)
    //丢弃无用的前导空格
    for i < n && s[i] == ' ' {
      i++
    }

  //标记正负号
  if i < n {
    if s[i] == '-' {
      sign = -1
      i++
    } else if s[i] == '+' {
      sign = 1
      i++
    }
  }

  // 统一处理正负数，均视为整数，只是最大值不同
  var max int64
  if sign > 0 {
    max = int64(math.MaxInt32)
  } else {
    max = int64(-math.MinInt32)
  }

  for i < n && s[i] >= '0' && s[i] <= '9' {
    abs = 10*abs + int64(s[i]-'0')
    if abs > max {
      abs = max
      break
    }

    i++
  }

  return int(sign * abs)
}

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    carry := 0
    const radix int = 10
    result := &ListNode{}
    p := result
    for l1 != nil || l2 != nil || carry != 0 {
        if l1 != nil {
            carry += l1.Val
            l1 = l1.Next
        }

        if l2 != nil {
            carry += l2.Val
            l2 = l2.Next
        }

        p.Next = &ListNode{
            Val: carry % radix,
            Next: nil,
        }
        p = p.Next
        carry /= radix
    }

    return result.Next
}
